LS jQuery Events Demo

.click()

$('.awesome-button').click(function( e ) {
	var $this   = $( this ),
		copyNum = +$this.data('num') + 1,
		btnMark = '<a class="awesome-button" href="#" data-num="' + copyNum + '">DYN BTN' + copyNum + '</a>';
 
	e.preventDefault();
 
	$this.closest('.parent-container')
		 .append( btnMark );
});
